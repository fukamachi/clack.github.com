Clack.App.URLMap - Map multiple apps in different paths.

SYNOPSIS

(defparameter *urlmap* (make-instance '<clack-app-urlmap>))  
(mount *urlmap* "/pc/" #'app-for-pc)  
(mount *urlmap* "/api/" #'app-for-api)  
(call *urlmap* env)  
 
;; Useful synonym.  
(builder-urlmap  
  ("/pc/" #'app-for-pc)  
  ("/api/" #'app-for-api))
